186  else if(wrap == Texture2DWrap::ClampToEdge)

189  return GL_CLAMP_TO_BORDER;
190  else

192  RL_CORE_ERROR("[Texture2D] No set any wrapping option. Rolled back to Texture2D::Repeat");
193  return GL_REPEAT;

195  };
